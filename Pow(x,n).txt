class Solution {
public:
    double myPow(double x, int n) {

        long long k = n;

        if(k < 0)
        {
            x = 1 / x;
            k = -k;
        }

        double p = 1;

        while(k > 0)
        {
            if(k % 2 == 1)
                p *= x;

            x *= x;
            k /= 2;
        }

        return p;
    }
};
